
## Sort of like git, but for tracking/creating/assigning scripts for local use

	fey init
		Initializes fey directory
	
	fey ls
		Lists commands
	
	fey add [command_name]
		Opens up an editor to let you type in the command
		[-f] allows you to override and edit existing commands
	
	fey add [command_name] ["script"]
		Assign the script to the command
		[-f] allows you to override and reassign existing commands
	
	fey rm [command_name]
	
	fey [command_name]
